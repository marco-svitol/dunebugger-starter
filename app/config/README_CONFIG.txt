# Dunebugger Starter - GPIO pin to remote IP configuration
# Update the natsServer setting with your target IP address
# 
# Examples:
# For a remote server at IP 192.168.1.100:
# natsServer = nats://192.168.1.100:4222
#
# For multiple servers:
# natsServer = nats://192.168.1.100:4222,nats://192.168.1.101:4222
#
# Default configuration (localhost for testing):
# natsServer = nats://localhost:4222
